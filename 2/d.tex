\begin{align*}
DF & = \frac{i_{1_{rms}}}{i_{rms}} \\
   & = \frac{\frac{4I_d}{\pi} \sin\left(\frac{\delta}{2}\right) \times \frac{1}{\sqrt{2}}}{I_d} \\
   & = \frac{2\sqrt{2}}{\pi} \sin\left( \frac{\delta}{2}\right) \\
   & = \frac{2\sqrt{2}}{\pi} \sin\left( \frac{30^\circ}{2}\right) \\
   & = 0.233
\end{align*}
