
% 1-wt
% 2 - alpha
% 3 - beta
\pgfmathdeclarefunction{truncSin}{3}{%
  \pgfmathparse{%
    (and(1,mod(#1,360)<=#3)*
    (and(1,mod(#1,360)>=#2)*
    (sin(#1))
    }%
}


% 1 - wt
% 2 - w
% 3 - R
% 4 - L
% 5 - phi
\pgfmathdeclarefunction{IL}{5}{%
  \pgfmathparse{%
     (
        sin((#1)-(#5))
        +sin(#5)*exp(-1*(#3)*((mod(#1,360)*0.02/360))/(#2*#4))
      )
    )
    }%
}
